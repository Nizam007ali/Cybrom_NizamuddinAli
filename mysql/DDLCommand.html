DDL commands


Create database;


create database student;



we must use database before start working

use student;



Creating table

create table Class11(id int, name varchar(20), DOB date,... , .......);



describe table schema

desc class11;



adding column to table respectively

alter table table_name add(per int, sal int);



delete column (only one at a time)

alter table table_name drop per;



modify column specification like datatype and size

alter table table_name modify name varchar(22);



Rename column

ALTER TABLE table_name RENAME COLUMN oldcolumn_name to newcolumn_name;



Rename table

RENAME TABLE COLUMN oldtable_name to newtable_name;



truncate use to delete whole table data

truncate table_name;

deleting table

drop table table_name;



deleting database

drop database database_name;
